<script>
  (function globalAttributionIdScope() {
    window.ATTRIBUTION_ID_GLOBAL_KEY = "tid";

    function setGlobalAttributionId() {
      const attributionIdFromQueryString = new URLSearchParams(
        window.location.search
      ).get(window.ATTRIBUTION_ID_GLOBAL_KEY);

      const attributionIdFromState = window.history.state
        ? window.history.state?.[window.ATTRIBUTION_ID_GLOBAL_KEY]
        : "";

      const referrer = document.referrer;

      if (attributionIdFromState) {
        window[window.ATTRIBUTION_ID_GLOBAL_KEY] = attributionIdFromState;
      } else if (
        attributionIdFromQueryString &&
        referrer.includes("typeform.com")
      ) {
        window[window.ATTRIBUTION_ID_GLOBAL_KEY] = attributionIdFromQueryString;
      }

      if (attributionIdFromQueryString) {
        const urlObj = new URL(window.location.href);
        urlObj.searchParams.delete(window.ATTRIBUTION_ID_GLOBAL_KEY);

        const urlWithoutAttributionId = urlObj.toString();

        window.history.replaceState(
          {
            ...window.history.state,
            [window.ATTRIBUTION_ID_GLOBAL_KEY]:
              window[window.ATTRIBUTION_ID_GLOBAL_KEY],
          },
          "",
          urlWithoutAttributionId
        );
      }
    }

    window.getAttributionUserId = function getAttributionUserId() {
      return window[window.ATTRIBUTION_ID_GLOBAL_KEY] || null;
    };
    window.setAttributionUserId = function setAttributionUserId(id) {
      window[window.ATTRIBUTION_ID_GLOBAL_KEY] = id;
    };

    function syncAttributionIdState() {
      const attributionUserIdFromCookie = Cookies.get("attribution_user_id");
      const attributionUserIdFromState = window.getAttributionUserId();
      /*
      1. If attribution cookie is present and different than global attribution id
      use attribution cookie as source of truth
    */
      if (
        attributionUserIdFromCookie &&
        attributionUserIdFromCookie !== attributionUserIdFromState
      ) {
        window.setAttributionUserId(attributionUserIdFromCookie);
      }

      /*
      2. If global attribution id is undefined/null create a new uuid
    */
      if (!window.getAttributionUserId()) {
        window.setAttributionUserId(uuidv4());
      }
      /*
      3. Finally, replace history state with the latest attribution id
      so that back/forward/refresh actions have the current state
    */
      window.history.replaceState(
        {
          ...window.history.state,
          [window.ATTRIBUTION_ID_GLOBAL_KEY]: window.getAttributionUserId(),
        },
        "",
        window.location.href
      );
    }

    setGlobalAttributionId();
    syncAttributionIdState();
  })();
</script>

      <h1 class="font-size-6xl font-normal mb-5" title="Create multi-language forms" itemprop="name">
        Create multi-language forms
        
      </h1>

      
      <section class="content article-content mb-6" itemprop="articleBody">
        
        <p><strong>Save time and reach a wider audience by creating a single form with multiple language options. You'll have the option to write your own translations or have AI translate your form for you.  Depending on your language settings, you can decide if you want to automatically have your form translated for respondents or give them the option to translate the form. The form will be translated if the respondent's browser language matches one of the translations you've added. All responses will be displayed in one place within your <a href="https://www.typeform.com/help/a/working-with-your-responses-the-table-view-360029253732/" target="_blank" rel="noopener noreferrer">Results</a> panel.</strong></p>
<p><strong><iframe src="//fast.wistia.com/embed/iframe/gxfitk4z21" width="560" height="315" frameborder="0" allowfullscreen=""></iframe></strong></p>
<p>You can create multi-language forms with the following languages:<strong><br></strong></p>
<ul>
<li>Arabic</li>
<li>Catalan</li>
<li>Chinese (simplified)</li>
<li>Chinese (traditional)</li>
<li>Croatian</li>
<li>Danish</li>
<li>Dutch</li>
<li>English</li>
<li>Estonian</li>
<li>Finnish</li>
<li>French</li>
<li>German (formal)</li>
<li>German (informal)</li>
<li>Greek</li>
<li>Hebrew</li>
<li>Hungarian</li>
<li>Italian</li>
<li>Japanese</li>
<li>Korean</li>
<li>Norwegian</li>
<li>Polish</li>
<li>Portuguese</li>
<li>Russian</li>
<li>Spanish</li>
<li>Swedish</li>
<li>Turkish</li>
<li>Ukrainian</li>
</ul>
<p>To create multi-language forms, you’ll need a <a href="https://www.typeform.com/pricing/" target="_blank" rel="noopener noreferrer">Business, Growth Pro, Growth Custom, or Enterprise plan</a>.</p>
<p>We recommend finalizing the text in your form before adding any translations, as this could cause your translations to not match any edited text. If you do need to edit your form, you can always go back and edit your translations to include any changed text.</p>
<h2 id="h_01HP2Q4WTJWH38DPZ5ANNEJ9K7">Add multiple languages to your form</h2>
<p><strong>1.</strong> Open your form and click the <strong>Translations</strong> icon.</p>
<p><img src="/hc/article_attachments/37056320859796" alt="Translate.png" width="625" height="323"></p>
<p><strong>2.</strong> Click <strong>+ Add languages</strong>.</p>
<p><img src="/hc/article_attachments/31444509441044" alt="2024-10-21_13-20-45.png" width="463" height="319"></p>
<p><strong>3.</strong> Select the languages you'd like to add to your form. Then click <strong>Add</strong> when you're done.</p>
<p>Respondents will only see the banner asking if they want to translate the form if their browser language matches one of the languages you've chosen for translations.</p>
<p><img src="/hc/article_attachments/31444515766676" alt="2024-10-21_13-23-25.png" width="462" height="394"></p>
<p><strong>4.</strong> Next, you'll see the status <strong>Translation needed</strong> for the language you selected.</p>
<p><img src="/hc/article_attachments/31444509442964" alt="2024-10-21_13-26-52.png" width="535" height="371"></p>
<p>Hover over the language you've added and you can choose how you'd like to translate your form:</p>
<p>You can <strong>Translate with AI</strong>.</p>
<p><img src="/hc/article_attachments/31444509444372" alt="2024-10-21_13-29-36.png" width="540" height="372"></p>
<p>Or <strong>Download a template</strong>.</p>
<p><img src="/hc/article_attachments/31444509447444" alt="2024-10-21_13-31-12.png" width="543" height="373"></p>
<p>In our example, we'll select <strong>Translate with AI</strong>. Then, if needed, we can edit the translations that were provided by AI.</p>
<p>After clicking Translate with AI, you'll see the status has changed to <strong>Translated</strong>.</p>
<p><img src="/hc/article_attachments/31444509448468" alt="2024-10-21_13-34-31.png" width="597" height="419"></p>
<p>If you're happy to use the translations provided by AI, you can click <strong>Publish</strong> and your form is ready to go. However, if you'd like to edit the translations provided by AI move on to the next section.</p>
<h2 id="h_01JT0SZQZ31906PTAKR668M5EM">Edit the translations provided by AI</h2>
<p><strong>1.</strong> Click <strong>Download template</strong>. This will download the latest stored translation in a CSV file.</p>
<p>In our example, it'll download the translations AI provided for Hungarian. </p>
<p><img src="/hc/article_attachments/31444509449748" alt="2024-10-21_13-42-38.png" width="596" height="427"></p>
<p><strong>2.</strong> Next, open the CSV file. We'll use Google Sheets to open our CSV file.</p>
<p><img src="/hc/article_attachments/31444515776276" alt="2024-10-21_13-48-25.png" width="625" height="317"></p>
<p>The CSV file contains the following 3 columns:</p>
<ul>
<li>
<strong>id</strong> - This column contains the ids of fields that have text in the form.</li>
<li>
<strong>original</strong> - This column lists the text in your form for questions, statements, answer options, and buttons in the original language. </li>
<li>
<strong>translation</strong> - This column lists the corresponding translations provided by AI.</li>
</ul>
<p><strong>3.</strong> Edit any of the translations in the <strong>translation</strong> column.</p>
<p><img src="/hc/article_attachments/31444515777300" alt="2024-10-21_17-18-54.png" width="625" height="275"></p>
<p>Here are a few things to keep in mind when editing translations:</p>
<ul>
<li>You can only edit text in the <strong>translation</strong> column. If you edit any of the other columns, you'll receive an error message when you upload the CSV back into Typeform.</li>
<li>All of the cells in the translations column must be completed. For example, if you have 15 cells in the <strong>original</strong> column with text, all 15 cells in the <strong>translation</strong> column must be filled in.</li>
<li>The number of rows must remain the same number from when you downloaded the template. You can't add additional rows to the CSV.</li>
</ul>
<p><strong>4.</strong> Download the edited translation template as a CSV file. </p>
<p><img src="/hc/article_attachments/31444515778452" alt="2024-10-21_17-26-51.png" width="625" height="344"></p>
<p><strong>5. </strong>Head back to your Typeform account and click <strong>Upload template</strong>. Only CSV files are supported.</p>
<p><img src="/hc/article_attachments/31444509455508" alt="2024-10-21_17-31-00.png" width="557" height="383"></p>
<p>A pop-up will appear to let you know that uploading a new translation will replace your current translation file. Click <strong>Upload translation</strong>.</p>
<p><img src="/hc/article_attachments/31444509456148" alt="2024-10-21_17-32-30.png" width="424" height="191"></p>
<p>Select the new CSV file you downloaded from step 8 and click <strong>Open</strong>.</p>
<p><img src="/hc/article_attachments/31444515782548" alt="2024-10-21_17-33-30.png" width="590" height="354"></p>
<p><strong>6.</strong> When you've finished uploading your new translations. Click <strong>Save </strong>or <strong>Publish edits</strong> to make your changes live.</p>
<p><img src="/hc/article_attachments/37056312986900" alt="Translate_publish_edits.png" width="625" height="323"></p>
<p>If you edit the text in your form in the Content panel after you've uploaded a translation, you'll see a notification letting you know that your form has changed and you'll need to update your translations.</p>
<p><img src="/hc/article_attachments/37056312987668" alt="Translate_message.png"></p>
<p>Click the <strong>Translations</strong> icon in the toolbar to update your translations. The status of your language will display as <strong>Update required</strong>. Select how you'd like to update your translations.</p>
<p><img src="/hc/article_attachments/31444509459348" alt="2024-10-21_17-43-41.png" width="524" height="359"></p>
<p>To delete a translation, click the trash can icon.</p>

<p>If you've made any edits or updates to your form, make sure you click <strong>Publish</strong> to make your changes live.</p>
<h2 id="01JMG0KZPQ19REYVBK0SW6K12A">Choose your language settings on how translations will be displayed</h2>
<p>By default, if a respondent's browser language matches one of the languages you've chosen for translation, the form will automatically be translated for them. </p>
<p>If you would like to give respondents the option to translate the form, they'll see a banner at the top of the form, asking if they want it to be translated. This banner will only appear if their browser language matches one of the languages you've chosen for translation. For instance, if your form is in English and you've selected Spanish as a translated language, Spanish-speaking respondents with their browser language set to Spanish will see the banner asking if they want the form to be translated into Spanish.</p>
<p>Choose your language settings:</p>
<p><strong>1.</strong> Open your form and click the settings icon in the tool bar.</p>
<p><img src="/hc/article_attachments/37056312989972" alt="Translate_settings.png"></p>
<p><strong>2.</strong> Select <strong>Language</strong>.</p>
<p><img src="/hc/article_attachments/37056312990868" alt="Translate_language.png" width="625" height="385"></p>
<p><strong>3.</strong> Under Translations, toggle <strong>Automatically translate to match browser language</strong> to be on or off. This is toggled on by default. Then click <strong>Save</strong> when done.</p>
<p><img src="/hc/article_attachments/37056320866708" alt="Transalte_auto.png" width="625" height="385"></p>
<p>If you've chosen to toggle off <strong>Automatically translate to match browser language</strong>, respondents will see a banner at the top of the screen asking them if they would like to translate the form to the language they have set for their browser.</p>
<p><img src="/hc/article_attachments/23541121557652" alt="multi-language07.png"></p>
<p>When respondents click the <strong>Translate</strong> button, the form will then appear in their browser's set language.</p>
<p><img src="/hc/article_attachments/23541121562644" alt="multi-language08.png"></p>
<h2 id="h_01HP2QK54YZ64VJZ2CTP80JBYJ">FAQ</h2>
<h3 id="h_01JK5PW62F29DTZHRF4WBMK059">How can I see the translations provided?</h3>
<p>You’ll be able to see the translations of your form if your browser's language is set to one of the translated languages you’ve selected for your form.<strong><br></strong></p>
<p>As a workaround, you can change the language settings of your browser to see the translations of your form. For example, if you’ve selected Spanish as one of the translated languages, you can set your browser’s language to Spanish and then you’ll be able to see the translations provided.</p>
<h3 id="h_01JK5PWCBRSBXQQ02CYSJWF3ZZ">Can I edit the translations?</h3>
<p>Yes, you can edit translations by downloading the CSV template. Then edit the translations in the translations columns and re-upload the CSV file back into Typeform. Advanced users can edit translations through our API. Check out our <a href="https://www.typeform.com/developers/create/translations/" target="_blank" rel="noopener noreferrer">API documentation here</a>.</p>
<p>Keep in mind that if you edit translations and then later <a href="/hc/en-us/articles/360029295012" target="_blank" rel="noopener noreferrer">duplicate</a> your form, the copy of your form won't contain the edited translations. </p>
<h3 id="h_01JK5PWJR78X2BTZ92773E29WC">How long will it take for my form to be translated?</h3>
<p>Translations can take about 10-15 seconds to load. If your form is very long, or the questions in your form are very wordy, it could take longer. Occasionally, translations may fail if the request is too large. <strong><br></strong></p>
<h3 id="h_01JK5PWRGAF6PZSQJ0BWDPRH4S">What language will my responses be shown in?</h3>
<p>Responses for close-ended question types (Multiple Choice, Dropdown, etc.) will be shown in the main language you’ve selected. Responses for open-ended question types (Short Text, Long Text, etc.) will be in the language respondents have entered their answers in.<strong><br></strong></p>
<h3 id="h_01JKV97C1SQ4GC739HJCVT4WZV">Are there any limitations with translating large forms?</h3>
<p>If you're using the option to translate your form with AI, there are a few limitations:</p>
<p>If your form's <a href="/hc/en-us/articles/360050914231" target="_blank" rel="noopener noreferrer">main language</a> is set to English, there are no limitations, unless you have one or more of the following in your form:</p>
<ul>
<li>The form has more than one hyperlinked URL. </li>
<li>The form has more than 100 answer options within one question, or if the form has a <a href="/hc/en-us/articles/360052316132" target="_blank" rel="noopener noreferrer">Question Group</a> with more than 100 answer options.</li>
<li>The form contains <a href="/hc/en-us/articles/360050914471" target="_blank" rel="noopener noreferrer">Variables</a> or <a href="/hc/en-us/articles/360052676372" target="_blank" rel="noopener noreferrer">URL parameters</a>.</li>
</ul>
<p>If your form contains one or more of the above items, we recommend following the limit suggestions for forms set to a non-English language.</p>
<p>If your form's main language is set to a non-English language, we recommend the following limits on your form:</p>
<ul>
<li>20 questions without answer options or up to 10 answer options from <a href="/hc/en-us/articles/360052409312" target="_blank" rel="noopener noreferrer">Multiple Choice</a>, <a href="/hc/en-us/articles/360037943011" target="_blank" rel="noopener noreferrer">Dropdown</a>, <a href="/hc/en-us/articles/360052865591" target="_blank" rel="noopener noreferrer">Picture Choice</a>, Yes/No, or Legal question types. </li>
<li>10 questions of any type with up to 20 answer options.</li>
<li>200 words in your form. This can be any text you've entered in your form, such as question text, labels for Picture Choice options, <a href="/hc/en-us/articles/360052366812" target="_blank" rel="noopener noreferrer">Welcome Screen</a>, etc.</li>
</ul>
<p>If you've exceeded the recommended limits listed above, this can cause an error when you generate translations with AI. You can however, download the CSV template of your form, manually enter all of the translations, and then re-upload the CSV.</p>
<h3 id="h_01JMMR9TG56JDF2D9M46BVE28B">How does multi-language forms work with embeds?</h3>
<p>Multi-language forms will work with forms that are <a href="https://help.typeform.com/hc/en-us/articles/360029249212-Embed-your-form" target="_blank" rel="noopener noreferrer">embedded on a website</a>. However, Multi-language forms won't work if you've <a href="https://help.typeform.com/hc/en-us/articles/360029664091-Launch-your-form-from-an-email" target="_blank" rel="noopener noreferrer">embedded the form in an email</a>.</p>
