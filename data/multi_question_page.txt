<script>
  (function globalAttributionIdScope() {
    window.ATTRIBUTION_ID_GLOBAL_KEY = "tid";

    function setGlobalAttributionId() {
      const attributionIdFromQueryString = new URLSearchParams(
        window.location.search
      ).get(window.ATTRIBUTION_ID_GLOBAL_KEY);

      const attributionIdFromState = window.history.state
        ? window.history.state?.[window.ATTRIBUTION_ID_GLOBAL_KEY]
        : "";

      const referrer = document.referrer;

      if (attributionIdFromState) {
        window[window.ATTRIBUTION_ID_GLOBAL_KEY] = attributionIdFromState;
      } else if (
        attributionIdFromQueryString &&
        referrer.includes("typeform.com")
      ) {
        window[window.ATTRIBUTION_ID_GLOBAL_KEY] = attributionIdFromQueryString;
      }

      if (attributionIdFromQueryString) {
        const urlObj = new URL(window.location.href);
        urlObj.searchParams.delete(window.ATTRIBUTION_ID_GLOBAL_KEY);

        const urlWithoutAttributionId = urlObj.toString();

        window.history.replaceState(
          {
            ...window.history.state,
            [window.ATTRIBUTION_ID_GLOBAL_KEY]:
              window[window.ATTRIBUTION_ID_GLOBAL_KEY],
          },
          "",
          urlWithoutAttributionId
        );
      }
    }

    window.getAttributionUserId = function getAttributionUserId() {
      return window[window.ATTRIBUTION_ID_GLOBAL_KEY] || null;
    };
    window.setAttributionUserId = function setAttributionUserId(id) {
      window[window.ATTRIBUTION_ID_GLOBAL_KEY] = id;
    };

    function syncAttributionIdState() {
      const attributionUserIdFromCookie = Cookies.get("attribution_user_id");
      const attributionUserIdFromState = window.getAttributionUserId();
      /*
      1. If attribution cookie is present and different than global attribution id
      use attribution cookie as source of truth
    */
      if (
        attributionUserIdFromCookie &&
        attributionUserIdFromCookie !== attributionUserIdFromState
      ) {
        window.setAttributionUserId(attributionUserIdFromCookie);
      }

      /*
      2. If global attribution id is undefined/null create a new uuid
    */
      if (!window.getAttributionUserId()) {
        window.setAttributionUserId(uuidv4());
      }
      /*
      3. Finally, replace history state with the latest attribution id
      so that back/forward/refresh actions have the current state
    */
      window.history.replaceState(
        {
          ...window.history.state,
          [window.ATTRIBUTION_ID_GLOBAL_KEY]: window.getAttributionUserId(),
        },
        "",
        window.location.href
      );
    }

    setGlobalAttributionId();
    syncAttributionIdState();
  })();
</script>

      <h1 class="font-size-6xl font-normal mb-5" title="Add a Multi-Question Page to your form" itemprop="name">
        Add a Multi-Question Page to your form
        
      </h1>

      
      <section class="content article-content mb-6" itemprop="articleBody">
        
        <p><strong>Traditionally, our forms have been designed to show one question at a time. This makes them feel more conversational and increases engagement. But sometimes, you might want to streamline your form by asking multiple questions on the same page. By adding a Multi-Question Page to your form, you can do just that.</strong></p>
<p><strong><iframe src="//fast.wistia.com/embed/iframe/64v4fnzmvo" width="560" height="315" frameborder="0" allowfullscreen=""></iframe></strong></p>
<p>When you add a Multi-Question page to your form, here's how questions will be displayed to your respondents.</p>
<p><img src="/hc/article_attachments/29638029607956" alt="2024-08-22_10-45-34.png"></p>
<h2 id="h_01JK873HDN9PA4KJ48PV6K7X68">Supported <a href="/hc/en-us/articles/360051789692" target="_blank" rel="noopener noreferrer">question types</a>
</h2>
<ul>
<li><strong>Checkbox</strong></li>
<li><strong>Dropdown</strong></li>
<li><strong>Email</strong></li>
<li><strong>Legal </strong></li>
<li><strong>Long Text</strong></li>
<li><strong>Multiple Choice</strong></li>
<li><strong>NPS</strong></li>
<li><strong>Opinion Scale</strong></li>
<li><strong>Phone Number</strong></li>
<li><strong>Ranking</strong></li>
<li><strong>Short Text</strong></li>
<li><strong>Statement</strong></li>
<li><strong>Yes/No</strong></li>
</ul>
<h2 id="h_01JK873TF8ACRYYV7T08PRW4M9">How to add a Multi-Question Page to your form</h2>
<p><strong>1.</strong> Go to the form you want to edit and click <strong>+ Add content</strong>.</p>
<p><img src="/hc/article_attachments/27895225584404" alt="2024-06-25_12-38-01.png"></p>
<p><strong>2. </strong>Select <strong>Multi-Question Page</strong> from the question menu.</p>
<p><img src="/hc/article_attachments/27894557082004" alt="2024-06-25_09-52-24.png"></p>
<p><strong>3.</strong> A new page will be added to your form where you can add as many questions as you like. To begin with, write your question header and a description. </p>
<p><img src="/hc/article_attachments/27895161754900" alt="2024-06-25_12-18-41.png"></p>
<div class="tipsnotes">
<h6 id="h_01J17FZ0D70BG3VVB9XMAHHD99" class="wysiwyg-text-align-left">
<img src="/hc/article_attachments/27704159088532" alt="HC_Note_new.png" width="120" height="106"><br><strong>Note</strong>! The question header of your Multi-Question Page won't be included in downloaded results. You'll only see results for each question within your Multi-Question Page. This is because the header doesn't collect results from your respondents.</h6>
</div>
<p><strong>4. </strong>Click the dropdown arrow next to your header. You’ll notice a Short Text question has been added automatically. </p>
<p><img src="/hc/article_attachments/27895129897108" alt="2024-06-25_12-20-54.png"></p>
<p>If you want to change the question type from Short Text to another question type, go to the question settings in the right-hand sidebar and choose a question type from the dropdown menu.</p>
<p><img src="/hc/article_attachments/29608606526356" alt="2024-08-21_15-37-19.png" width="290" height="464"></p>
<p><strong>5.</strong> Write your first question.</p>
<p> </p>
<p><strong><img src="/hc/article_attachments/27895161761684" alt="2024-06-25_12-26-09 (1).gif"></strong></p>
<p><strong>6.</strong> Then click <strong>+ Add content</strong> to add another question. </p>
<p><img src="/hc/article_attachments/27895129904532" alt="2024-06-25_12-28-52.png"></p>
<p><strong>7.</strong> You can change your question settings in the right-hand sidebar. For example, depending on your question type, you can make the question required or set a character limit for responses.</p>
<p><img src="/hc/article_attachments/27894525063188" alt="2024-05-30_12-17-23.png" width="338" height="441"></p>
<p><strong>8. </strong>Once you’ve added all the questions you want, you’re done!</p>
<p>If you want to add more questions to your form outside of the Multi-Question page, click <strong>+ Add content</strong> and select a question type.</p>
<p>If you select a question type that's compatible with the Multi-Question Page (see the top of this article for a full list of supported question types), it'll be placed in the Multi-Question Page by default, but you can move it by clicking and dragging it out of the Multi-Question Page.</p>
<p><img src="/hc/article_attachments/27895161765524" alt="2024-06-25_12-31-45.png"></p>
<p>You can also click on the question type icons and drag them around to change the order within the Multi-Question Page.</p>
<p>Similarly, you can drag and drop compatible question types into your Multi-Question Page.</p>
<p><img src="/hc/article_attachments/27894705132820" alt="2024-06-25_10-14-54 (1).gif"></p>
<p><strong>9.</strong> If you want to add logic to your form, head to the <strong>Logic</strong> panel to add <a href="https://help.typeform.com/hc/en-us/articles/360054770931-Use-Branching-and-calculations-in-Logic-to-show-relevant-questions" target="_blank" rel="noopener noreferrer">Branching</a> rules and conditions to your Multi-Question Page.</p>
<p><img src="/hc/article_attachments/34978560379028" alt="Screenshot 2025-02-11 at 14.43.03.png"></p>
<p>You can select the individual questions within your Multi-Question page to send your respondents down different paths or to segment your audience.</p>
<p><img src="/hc/article_attachments/34978560380436" alt="Screenshot 2025-02-11 at 16.40.31.png"></p>
<p>And voilà! Here’s how our Multi-Question Page will look to our respondents.</p>
<p><img src="/hc/article_attachments/29638029607956" alt="2024-08-22_10-45-34.png"></p>
<p><strong>10.</strong> Once you’ve collected some responses, go to the <strong><a href="/hc/en-us/articles/360029253732" target="_blank" rel="noopener noreferrer">Results</a> </strong>panel to see your data. Answers to each question on a Multi-Question Page will be shown as standalone answers in your results, just like in forms with only one question per page.</p>
<p><img src="/hc/article_attachments/29638029609620" alt="2024-08-22_10-51-34.png"></p>
<p><strong>11. </strong>Similarly, when you connect a form containing a Multi-Question Page to another tool, it works just like with any other question type. You can map each question from the Multi-Question Page to a different field in the integration.</p>
<p><img src="/hc/article_attachments/29638133653652" alt="2024-08-22_10-54-05.png" width="604" height="410"></p>
<p>Check out all our integrations <a href="https://www.google.com/url?q=https://www.typeform.com/help/s/integrations-360005963612/&amp;sa=D&amp;source=docs&amp;ust=1718806050797680&amp;usg=AOvVaw2IrQgSWyAbcl9eTy0bVh9g" target="_blank" rel="noopener noreferrer">here</a>.</p>
<h2 id="h_01J5WJ891RV7HPM0S6FYHSGCZJ">FAQ</h2>
<h4 id="h_01J5WNKGE8ZJZB2XDYK4R8Z7SR"><strong>Can I add a <a href="https://www.typeform.com/help/a/collect-partial-responses-21102221958676/" target="_blank" rel="noopener noreferrer">Partial Submit Point</a> after the Multi-Question Page?</strong></h4>
<p>You can add a <a href="https://www.typeform.com/help/a/collect-partial-responses-21102221958676/" target="_blank" rel="noopener noreferrer">Partial Submit Point</a> after the Multi-Question Page, just like with any other question type. However, you can't add a Partial Submit Point after a specific question within your Multi-Question Page.</p>
<h4 id="h_01J5WNKP9HFH6SCRCH114Y356S"><strong>Does the Multi-Question Page support <a href="https://www.typeform.com/help/a/lead-enrichment-and-qualification-23794452273812/" target="_blank" rel="noopener noreferrer">Data Enrichment</a>?</strong></h4>
<p>Yes. You can collect <a href="https://www.typeform.com/help/a/lead-enrichment-and-qualification-23794452273812/" target="_blank" rel="noopener noreferrer">enrichment data</a> with an email question in the Multi-Question Page if it is the first email question of your form.</p>
<h4 id="h_01J5WNKVABAB247M2BV4XR8XE8"><strong>Can I use <a href="https://www.typeform.com/help/a/what-is-logic-360029116392/" target="_blank" rel="noopener noreferrer">Logic</a> with the Multi-Question Page?</strong></h4>
<p>Yes. You can use <a href="https://www.typeform.com/help/a/what-is-logic-360029116392/" target="_blank" rel="noopener noreferrer">Logic</a> to direct respondents to a Multi-Question page or use answers provided in the Multi-Question Page in Logic rules.</p>
<h4 id="h_01J8365BDAMZ0WE4ERMYC4DA6B"><strong>Can I add a <a href="/hc/en-us/articles/360054424131" target="_blank" rel="noopener noreferrer">score</a> or create an <a href="/hc/en-us/articles/360058072351" target="_blank" rel="noopener noreferrer">outcome quiz</a> with the Multi-Question Page?</strong></h4>
<p>No. Currently, adding a score or creating an outcome quiz is not supported with the Multi-Question Page.</p>
<p>If you want to create an umbrella question with sub-questions on separate pages, check out our <a href="/hc/en-us/articles/360052316132" target="_blank" rel="noopener noreferrer">Question Group</a> feature.</p>
<p><strong>Can I embed a Multi-Question Page in an email?</strong></p>
<p>No. <a href="https://help.typeform.com/hc/en-us/articles/360029664091-Launch-your-form-from-an-email" target="_blank" rel="noopener noreferrer">Launching your form in an email</a> with the Multi-Question Page as your first question is currently not supported.</p>
